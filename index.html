                    `;

                });

                

                html += '</div>';

            });

            

            container.innerHTML = html;

        }


        // Calculate standings

        function calculateStandings() {

            const standings = {};

            

            Object.keys(allMatches).forEach(groupKey => {

                const matches = allMatches[groupKey];

                const players = tournamentData.groups[groupKey].players;

                

                standings[groupKey] = {};

                players.forEach(player => {

                    standings[groupKey][player.name] = {

                        name: player.name,

                        wins: 0,

                        losses: 0,

                        played: 0

                    };

                });

                

                matches.forEach(match => {

                    if (match.result) {

                        const winner = match.result.winner;

                        const loser = winner === match.player1.name ? match.player2.name : match.player1.name;

                        

                        standings[groupKey][winner].wins++;

                        standings[groupKey][winner].played++;

                        standings[groupKey][loser].losses++;

                        standings[groupKey][loser].played++;

                    }

                });

            });

            

            return standings;

        }


        // Render standings tab

        function renderStandings() {

            const container = document.getElementById('standings-content');

            const standings = calculateStandings();

            

            let html = '<div class="standings-grid">';

            

            Object.keys(standings).forEach(groupKey => {

                const group = tournamentData.groups[groupKey];

                const groupStandings = Object.values(standings[groupKey])

                    .sort((a, b) => {

                        if (b.wins !== a.wins) return b.wins - a.wins;

                        return a.losses - b.losses;

                    });

                

                html += `

                    <div class="standings-group">

                        <h3>${group.name}</h3>

                `;

                

                groupStandings.forEach((player, index) => {

                    const isQualified = index < 2;

                    html += `

                        <div class="standings-row ${isQualified ? 'qualified' : ''}">

                            <div style="display: flex; align-items: center;">

                                <span class="position">${index + 1}.</span>

                                <span class="standings-name">${player.name}</span>

                            </div>

                            <div class="standings-stats">

                                <span class="wins">${player.wins}W</span>

                                <span class="losses">${player.losses}L</span>

                            </div>

                        </div>

                    `;

                });

                

                html += '</div>';

            });

            

            html += '</div>';

            container.innerHTML = html;

        }


        // Modal functions

        function openModal(matchId) {

            const [groupKey, matchIndex] = matchId.split('-');

            const match = allMatches[groupKey][matchIndex];

            currentMatch = { groupKey, matchIndex, match };

            

            const modal = document.getElementById('recordModal');

            const winnerSelect = document.getElementById('winner');

            const marginSelect = document.getElementById('margin');

            

            // Populate winner dropdown

            winnerSelect.innerHTML = `

                <option value="">Select winner...</option>

                <option value="${match.player1.name}">${match.player1.name}</option>

                <option value="${match.player2.name}">${match.player2.name}</option>

            `;

            

            // If editing existing result, pre-populate

            if (match.result) {

                winnerSelect.value = match.result.winner;

                marginSelect.value = match.result.margin;

            } else {

                winnerSelect.value = '';

                marginSelect.value = '';

            }

            

            modal.classList.add('active');

        }


        function closeModal() {

            document.getElementById('recordModal').classList.remove('active');

            document.getElementById('recordForm').reset();

            currentMatch = null;

        }


        // Handle form submission

        document.getElementById('recordForm').addEventListener('submit', function(e) {

            e.preventDefault();

            

            const winner = document.getElementById('winner').value;

            const margin = document.getElementById('margin').value;

            

            if (winner && margin && currentMatch) {

                // Update Firebase

                const matchRef = database.ref(`matches/${currentMatch.groupKey}/${currentMatch.matchIndex}/result`);

                matchRef.set({

                    winner: winner,

                    margin: margin

                });

                

                closeModal();

            }

        });


        // Tab switching

        function showTab(tab) {

            // Update tab buttons

            document.querySelectorAll('.tab').forEach(btn => btn.classList.remove('active'));

            event.target.classList.add('active');

            

            // Update content

            document.querySelectorAll('.tab-content').forEach(content => content.style.display = 'none');

            document.getElementById(`${tab}-content`).style.display = 'block';

            

            renderCurrentTab();

        }


        function renderCurrentTab() {

            const activeTab = document.querySelector('.tab.active').textContent.toLowerCase();

            if (activeTab === 'matches') {

                renderMatches();

            } else if (activeTab === 'standings') {

                renderStandings();

            }

        }
